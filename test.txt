private void printFillable(ClaimStatusData claimStatusDataResponse) {
    // Safely handle possible nulls using Optional.
    Optional.ofNullable(claimStatusDataResponse)
        .map(ClaimStatusData::getClaimStatusDetailsList) // get the list
        .ifPresent(claimStatusDetailsList -> 
            // Stream over the list
            claimStatusDetailsList.stream().forEach(claimStatusDetails -> {
                // Print the templateId
                String templateId = claimStatusDetails.getTemplateId();
                logger.debug("Fillable values for template ID : {}", templateId);

                // Iterate over the map entries
                Map<String, Object> items = claimStatusDetails.getTemplateMap();
                if (items != null) {
                    items.forEach((key, value) -> 
                        logger.debug("Key: {}, Value: {}", key, value)
                    );
                }
            })
        );
}
